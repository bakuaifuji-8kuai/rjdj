SectionManage.vue - 分段管理

/workspace/src/views/tunnel/SectionManage.vue 
<template>
  <div class="page-container">
    <div class="page-header">
      <el-button type="primary" @click="handleAdd">新增分段</el-button>
      <el-button type="danger" @click="handleBatchDelete">批量删除</el-button>
    </div>
    <el-table :data="sectionList" border @selection-change="handleSelection">
      <el-table-column type="selection"/>
      <el-table-column prop="no" label="序号"/>
      <el-table-column prop="name" label="分段名称"/>
      <el-table-column prop="start" label="起始位置"/>
      <el-table-column prop="end" label="结束位置"/>
      <el-table-column label="操作">
        <template #default="{row}">
          <el-button size="small" @click="handleEdit(row)">修改</el-button>
          <el-button size="small" type="danger" @click="handleDelete(row)">删除</el-button>
        </template>
      </el-table-column>
    </el-table>
  </div>
</template>
<script setup>
import { ref } from 'vue'
import { ElMessage, ElMessageBox } from 'element-plus'

const sectionList = ref([
  { no: 1, name: '入口段', start: '0m', end: '300m' },
  { no: 2, name: '过渡段', start: '300m', end: '600m' },
  { no: 3, name: '中间段', start: '600m', end: '2200m' },
  { no: 4, name: '出口段', start: '2200m', end: '2800m' }
])
const selectedRows = ref([])

const handleSelection = (val) => { selectedRows.value = val }
const handleAdd = () => { ElMessage.info('新增分段') }
const handleEdit = (row) => { ElMessage.info('修改分段: ' + row.name) }
const handleDelete = (row) => {
  ElMessageBox.confirm('确定删除分段【' + row.name + '】?')
    .then(() => {
      sectionList.value = sectionList.value.filter(s => s.no !== row.no)
      ElMessage.success('删除成功')
    })
}
const handleBatchDelete = () => {
  if(selectedRows.value.length === 0) {
    ElMessage.warning('请选择要删除的分段')
    return
  }
  ElMessageBox.confirm('确定删除选中的 ' + selectedRows.value.length + ' 个分段?')
    .then(() => {
      const ids = selectedRows.value.map(s => s.no)
      sectionList.value = sectionList.value.filter(s => !ids.includes(s.no))
      selectedRows.value = []
      ElMessage.success('批量删除成功')
    })
}
</script>
